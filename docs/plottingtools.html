128def multiplot(
129        nrows: int,
130        ncols: int,
131        size_xy: Tuple[float, float],
132        wspace: Optional[float] = None,
133        hspace: Optional[float] = None) -> Tuple[matplotlib.figure.Figure, plt.Axes]:
134    """Generates a new figure consisting of nrows rows and ncols columns of plots with overall figure size size_xy. Horizontal and vertical distance between plots may be defined explicitly.
135    
136    Args:
137        size:
138            An tuple of two floats, containing the desired figure width and heigth in inches.
139        wspace:
140            An optional float, specifying the horizontal distance between columns. Defaults to zero.
141        hspace:
142            An optional float, specifying the vertical distance between rows. Defaults to zero.
143    
144    Returns:
145        fig:
146            A matplotlib.figure.Figure instance
147        ax:
148            An array of pyplot.Axes instances
149    """    
150    
151    assert type(nrows) == int
152    assert type(ncols) == int
153    assert nrows > 0
154    assert ncols > 0
155
156    try:
157        l, h = size_xy
158        l = float(l)
159        h = float(h)
160        if hspace is not None:
161            hspace = float(hspace)
162        if wspace is not None:
163            wspace = float(wspace)
164    except:
165        raise AssertionError
166    assert l > 0
167    assert h > 0
168
169    fig, ax = plt.subplots(nrows, ncols, figsize=(l, h))
170    if hspace is not None:
171        plt.subplots_adjust(hspace=hspace)
172    if wspace is not None:
173        plt.subplots_adjust(wspace=wspace)
174    return fig, ax

294    assert hasattr(ax, 'plot')

304def rectangle(ax: plt.Axes,
305              x1: float,
306              y1: float,
307              x2: float,
308              y2: float,
309              **kwargs) -> None:
310    """Convenience function for addig a rectangle to an existing plot without having to manually call ax.add_patch(). Takes x and y coordinates of two points as arguments, instead of the x and y coordinate of one point and the rectagle width and heigth, like add_patch() would.
311    
312    Args:
313        ax:
314            A pyplot.Axes instance
315        x1:
316            A float, specifying the x coordinate of the first point.
317        y1:
318            A float, specifying the y coordinate of the first point.
319        x2:
320            A float, specifying the x coordinate of the second point.
321        y2:
322            A float, specifying the y coordinate of the second point.
323        **kwargs:
324            Named arguments such as color, fill, linewidth, linestyle. Passed to ax.add_patch().
325    
326    Returns:
327        None
328    """
329    try:
330        x1 = float(x1)
331        y1 = float(y1)
332        x2 = float(x2)
333        y2 = float(y2)
334    except:
335        raise AssertionError
336    assert hasattr(ax, 'plot')
337
338    ax.add_patch(
339        Rectangle((x1, y1),
340                  x2 - x1,
341                  y2 - y1,
342                  **kwargs))

438def ticks_and_labels(ax: plt.Axes,
439                     which: str,
440                     ticks: List[float],
441                     labels: Optional[List[str]] = None) -> None:
442    try:
443        ticks = list(ticks)
444        ticks = [float(t) for t in ticks]
445    except:
446        raise AssertionError
447
448    assert which in ["x", "y", "xy", "yx"]
449    assert hasattr(ax, 'plot')
450
451    if labels is None:
452        labels = list(map(lambda x: str(x), ticks))
453    else:
454        try:
455            labels = list(labels)
456        except:
457            raise AssertionError
458
459        assert len(ticks) == len(labels)
460
461    if which == "x" or which == "xy" or which == "yx":
462        ax.set_xticks(ticks)
463        ax.set_xticklabels(labels)
464    if which == "y" or which == "xy" or which == "yx":
465        ax.set_yticks(ticks)
466        ax.set_yticklabels(labels)

485def align_ticklabels(ax: plt.Axes,
486                     which: str,
487                     horizontal: Optional[str] = None,
488                     vertical: Optional[str] = None) -> None:
489    assert which in ["x", "y"]
490    assert hasattr(ax, 'plot')
491
492    if which == "x":
493        if horizontal is not None:
494            assert horizontal in ['center', 'right', 'left']
495            ax.set_xticklabels(ax.get_xticklabels(),
496                               horizontalalignment=horizontal)
497        if vertical is not None:
498            assert vertical in ['center', 'top', 'bottom', 'baseline']
499            ax.set_xticklabels(ax.get_xticklabels(),
500                               verticalalignment=vertical)
501
502    if which == "y":
503        if horizontal is not None:
504            assert horizontal in ['center', 'right', 'left']
505            ax.set_yticklabels(ax.get_yticklabels(),
506                               horizontalalignment=horizontal)
507        if vertical is not None:
508            assert vertical in ['center', 'top', 'bottom', 'baseline']
509            ax.set_yticklabels(ax.get_yticklabels(),
510                               verticalalignment=vertical)
